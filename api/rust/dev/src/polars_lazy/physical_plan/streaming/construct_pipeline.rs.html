45fn to_physical_piped_expr(
46    expr: &ExprIR,
47    expr_arena: &Arena<AExpr>,
48    schema: &SchemaRef,
49) -> PolarsResult<Arc<dyn PhysicalPipedExpr>> {
50    // this is a double Arc<dyn> explore if we can create a single of it.

61fn jit_insert_slice(
62    node: Node,
63    lp_arena: &mut Arena<IR>,
64    sink_nodes: &mut Vec<(usize, Node, Rc<RefCell<u32>>)>,
65    operator_offset: usize,
66) {
67    // if the join has a slice, we add a new slice node
68    // note that we take the offset + 1, because we want to
69    // slice AFTER the join has happened and the join will be an
70    // operator
71    // NOTE: Don't do this for union, that doesn't work.
72    // TODO! Deal with this in the optimizer.
73    use IR::*;
74    let (offset, len) = match lp_arena.get(node) {
75        Join { options, .. } if options.args.slice.is_some() => {
76            let Some((offset, len)) = options.args.slice else {
77                unreachable!()
78            };
79            (offset, len)
80        },
81        _ => return,
82    };
83
84    let slice_node = lp_arena.add(Slice {
85        input: node,
86        offset,
87        len: len as IdxSize,
88    });
89    sink_nodes.push((operator_offset + 1, slice_node, Rc::new(RefCell::new(1))));
90}

92pub(super) fn construct(
93    tree: Tree,
94    lp_arena: &mut Arena<IR>,
95    expr_arena: &mut Arena<AExpr>,
96    fmt: bool,
97) -> PolarsResult<Option<Node>> {
98    use IR::*;
99
100    let mut pipelines = Vec::with_capacity(tree.len());
101    let mut callbacks = CallBacks::new();
102
103    let is_verbose = verbose();
104
105    // First traverse the branches and nodes to determine how often a sink is
106    // shared.
107    // This shared count will be used in the pipeline to determine
108    // when the sink can be finalized.
109    let mut sink_share_count = PlHashMap::new();
110    let n_branches = tree.len();
111    if n_branches > 1 {
112        for branch in &tree {
113            for op in branch.operators_sinks.iter() {
114                match op {
115                    PipelineNode::Sink(sink) => {
116                        let count = sink_share_count
117                            .entry(sink.0)
118                            .or_insert(Rc::new(RefCell::new(0u32)));
119                        *count.borrow_mut() += 1;
120                    },
121                    PipelineNode::RhsJoin(node) => {
122                        let _ = callbacks.insert(*node, get_dummy_operator());
123                    },
124                    _ => {},
125                }
126            }
127        }
128    }
129
130    // Shared sinks are stored in a cache, so that they share state.
131    // If the shared sink is already in cache, that one is used.
132    let mut sink_cache = PlHashMap::new();
133    let mut final_sink = None;
134
135    for branch in tree {
136        // The file sink is always to the top of the tree
137        // not every branch has a final sink. For instance rhs join branches
138        if let Some(node) = branch.get_final_sink() {
139            if matches!(lp_arena.get(node), IR::Sink { .. }) {
140                final_sink = Some(node)
141            }
142        }
143        // should be reset for every branch
144        let mut sink_nodes = vec![];
145
146        let mut operators = Vec::with_capacity(branch.operators_sinks.len());
147        let mut operator_nodes = Vec::with_capacity(branch.operators_sinks.len());
148
149        // iterate from leaves upwards
150        let mut iter = branch.operators_sinks.into_iter().rev();
151
152        for pipeline_node in &mut iter {
153            let operator_offset = operators.len();
154            match pipeline_node {
155                PipelineNode::Sink(node) => {
156                    let shared_count = if n_branches > 1 {
157                        // should be here
158                        sink_share_count.get(&node.0).unwrap().clone()
159                    } else {
160                        Rc::new(RefCell::new(1))
161                    };
162                    sink_nodes.push((operator_offset, node, shared_count))
163                },
164                PipelineNode::Operator(node) => {
165                    operator_nodes.push(node);
166                    let op = get_operator(node, lp_arena, expr_arena, &to_physical_piped_expr)?;
167                    operators.push(op);
168                },
169                PipelineNode::Union(node) => {
170                    operator_nodes.push(node);
171                    let op = get_operator(node, lp_arena, expr_arena, &to_physical_piped_expr)?;
172                    operators.push(op);
173                },
174                PipelineNode::RhsJoin(node) => {
175                    operator_nodes.push(node);
176                    jit_insert_slice(node, lp_arena, &mut sink_nodes, operator_offset);
177                    let op = callbacks.get(&node).unwrap().clone();
178                    operators.push(Box::new(op))
179                },
180            }
181        }
182
183        let pipeline = create_pipeline(
184            &branch.sources,
185            operators,
186            sink_nodes,
187            lp_arena,
188            expr_arena,
189            to_physical_piped_expr,
190            is_verbose,
191            &mut sink_cache,
192            &mut callbacks,
193        )?;
194        pipelines.push(pipeline);
195    }
196
197    let Some(final_sink) = final_sink else {
198        return Ok(None);
199    };
200    let insertion_location = match lp_arena.get(final_sink) {
201        // this was inserted only during conversion and does not exist
202        // in the original tree, so we take the input, as that's where
203        // we connect into the original tree.
204        Sink {
205            input,
206            payload: SinkTypeIR::Memory,
207        } => *input,
208        // Other sinks were not inserted during conversion,
209        // so they are returned as-is
210        Sink { .. } => final_sink,
211        _ => unreachable!(),
212    };
213    // keep the original around for formatting purposes
214    let original_lp = if fmt {
215        let original_lp = IRPlan::new(insertion_location, lp_arena.clone(), expr_arena.clone());
216        Some(original_lp)
217    } else {
218        None
219    };
220
221    // Replace the part of the logical plan with a `MapFunction` that will execute the pipeline.
222    let schema = lp_arena
223        .get(insertion_location)
224        .schema(lp_arena)
225        .into_owned();
226    let pipeline_node = get_pipeline_node(lp_arena, pipelines, schema, original_lp);
227    lp_arena.replace(insertion_location, pipeline_node);
228
229    Ok(Some(final_sink))
230}

232fn get_pipeline_node(
233    lp_arena: &mut Arena<IR>,
234    mut pipelines: Vec<PipeLine>,
235    schema: SchemaRef,
236    original_lp: Option<IRPlan>,
237) -> IR {
238    // create a dummy input as the map function will call the input
239    // so we just create a scan that returns an empty df
240    let dummy = lp_arena.add(IR::DataFrameScan {
241        df: Arc::new(DataFrame::empty()),
242        schema: Arc::new(Schema::default()),
243        output_schema: None,
244    });
245
246    IR::MapFunction {
247        function: FunctionIR::Pipeline {
248            function: Arc::new(Mutex::new(move |_df: DataFrame| {
249                let state = ExecutionState::new();
250                if state.verbose() {
251                    eprintln!("RUN STREAMING PIPELINE");
252                    eprintln!("{:?}", &pipelines)
253                }
254                execute_pipeline(state, std::mem::take(&mut pipelines))
255            })),
256            schema,
257            original: original_lp.map(Arc::new),
258        },
259        input: dummy,
260    }
261}
