9    /// Get the inner data type of the list.
10    pub fn inner_dtype(&self) -> &DataType {
11        match self.dtype() {
12            DataType::List(dt) => dt.as_ref(),
13            _ => unreachable!(),
14        }
15    }

17    pub fn set_inner_dtype(&mut self, dtype: DataType) {
18        assert_eq!(dtype.to_physical(), self.inner_dtype().to_physical());
19        let field = Arc::make_mut(&mut self.field);
20        field.coerce(DataType::List(Box::new(dtype)));
21    }

31    /// Set the logical type of the [`ListChunked`].
32    ///
33    /// # Safety
34    /// The caller must ensure that the logical type given fits the physical type of the array.
35    pub unsafe fn to_logical(&mut self, inner_dtype: DataType) {
36        debug_assert_eq!(&inner_dtype.to_physical(), self.inner_dtype());
37        let fld = Arc::make_mut(&mut self.field);
38        fld.coerce(DataType::List(Box::new(inner_dtype)))
39    }

41    /// Convert the datatype of the list into the physical datatype.
42    pub fn to_physical_repr(&self) -> Cow<ListChunked> {
43        let Cow::Owned(physical_repr) = self.get_inner().to_physical_repr() else {
44            return Cow::Borrowed(self);
45        };
46
47        let ca = if physical_repr.chunks().len() == 1 && self.chunks().len() > 1 {
48            // Physical repr got rechunked, rechunk self as well.
49            self.rechunk()
50        } else {
51            Cow::Borrowed(self)
52        };
53
54        assert_eq!(ca.chunks().len(), physical_repr.chunks().len());
55
56        let chunks: Vec<_> = ca
57            .downcast_iter()
58            .zip(physical_repr.into_chunks())
59            .map(|(chunk, values)| {
60                LargeListArray::new(
61                    ArrowDataType::LargeList(Box::new(ArrowField::new(
62                        PlSmallStr::from_static("item"),
63                        values.dtype().clone(),
64                        true,
65                    ))),
66                    chunk.offsets().clone(),
67                    values,
68                    chunk.validity().cloned(),
69                )
70                .to_boxed()
71            })
72            .collect();
73
74        let name = self.name().clone();
75        let dtype = DataType::List(Box::new(self.inner_dtype().to_physical()));
76        Cow::Owned(unsafe { ListChunked::from_chunks_and_dtype_unchecked(name, chunks, dtype) })
77    }

79    /// Convert a non-logical [`ListChunked`] back into a logical [`ListChunked`] without casting.
80    ///
81    /// # Safety
82    ///
83    /// This can lead to invalid memory access in downstream code.
84    pub unsafe fn from_physical_unchecked(
85        &self,
86        to_inner_dtype: DataType,
87    ) -> PolarsResult<ListChunked> {
88        debug_assert!(!self.inner_dtype().is_logical());
89
90        let inner_chunks = self
91            .downcast_iter()
92            .map(|chunk| chunk.values())
93            .cloned()
94            .collect();
95
96        let inner = unsafe {
97            Series::from_chunks_and_dtype_unchecked(
98                PlSmallStr::EMPTY,
99                inner_chunks,
100                self.inner_dtype(),
101            )
102        };
103        let inner = unsafe { inner.from_physical_unchecked(&to_inner_dtype) }?;
104
105        let chunks: Vec<_> = self
106            .downcast_iter()
107            .zip(inner.into_chunks())
108            .map(|(chunk, values)| {
109                LargeListArray::new(
110                    ArrowDataType::LargeList(Box::new(ArrowField::new(
111                        PlSmallStr::from_static("item"),
112                        values.dtype().clone(),
113                        true,
114                    ))),
115                    chunk.offsets().clone(),
116                    values,
117                    chunk.validity().cloned(),
118                )
119                .to_boxed()
120            })
121            .collect();
122
123        let name = self.name().clone();
124        let dtype = DataType::List(Box::new(to_inner_dtype));
125        Ok(unsafe { ListChunked::from_chunks_and_dtype_unchecked(name, chunks, dtype) })
126    }

128    /// Get the inner values as [`Series`], ignoring the list offsets.
129    pub fn get_inner(&self) -> Series {
130        let chunks: Vec<_> = self.downcast_iter().map(|c| c.values().clone()).collect();
131
132        // SAFETY: Data type of arrays matches because they are chunks from the same array.
133        unsafe {
134            Series::from_chunks_and_dtype_unchecked(self.name().clone(), chunks, self.inner_dtype())
135        }
136    }

138    /// Ignore the list indices and apply `func` to the inner type as [`Series`].
139    pub fn apply_to_inner(
140        &self,
141        func: &dyn Fn(Series) -> PolarsResult<Series>,
142    ) -> PolarsResult<ListChunked> {
143        // generated Series will have wrong length otherwise.
144        let ca = self.rechunk();
145        let arr = ca.downcast_as_array();
146
147        // SAFETY:
148        // Inner dtype is passed correctly
149        let elements = unsafe {
150            Series::from_chunks_and_dtype_unchecked(
151                self.name().clone(),
152                vec![arr.values().clone()],
153                ca.inner_dtype(),
154            )
155        };
156
157        let expected_len = elements.len();
158        let out: Series = func(elements)?;
159        polars_ensure!(
160            out.len() == expected_len,
161            ComputeError: "the function should apply element-wise, it removed elements instead"
162        );
163        let out = out.rechunk();
164        let values = out.chunks()[0].clone();
165
166        let inner_dtype = LargeListArray::default_datatype(values.dtype().clone());
167        let arr = LargeListArray::new(
168            inner_dtype,
169            (*arr.offsets()).clone(),
170            values,
171            arr.validity().cloned(),
172        );
173
174        // SAFETY: arr's inner dtype is derived from out dtype.
175        Ok(unsafe {
176            ListChunked::from_chunks_and_dtype_unchecked(
177                ca.name().clone(),
178                vec![Box::new(arr)],
179                DataType::List(Box::new(out.dtype().clone())),
180            )
181        })
182    }
