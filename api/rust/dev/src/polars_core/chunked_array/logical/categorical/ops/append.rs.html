1use polars_error::constants::LENGTH_LIMIT_MSG;

3use super::*;
4use crate::chunked_array::ops::append::new_chunks;

6struct CategoricalAppend;
7
8impl CategoricalMergeOperation for CategoricalAppend {
9    fn finish(self, lhs: &UInt32Chunked, rhs: &UInt32Chunked) -> PolarsResult<UInt32Chunked> {
10        let mut lhs_mut = lhs.clone();
11        lhs_mut.append(rhs)?;
12        Ok(lhs_mut)
13    }
14}
15
16impl CategoricalChunked {
17    fn set_lengths(&mut self, other: &Self) {
18        let length_self = &mut self.physical_mut().length;
19        *length_self = length_self
20            .checked_add(other.len())
21            .expect(LENGTH_LIMIT_MSG);
22
23        assert!(
24            IdxSize::try_from(*length_self).is_ok(),
25            "{}",
26            LENGTH_LIMIT_MSG
27        );
28        self.physical_mut().null_count += other.null_count();
29    }
30
31    pub fn append(&mut self, other: &Self) -> PolarsResult<()> {
32        polars_ensure!(!self.is_enum() || self.dtype() == other.dtype(), append);
33
34        // fast path all nulls
35        if self.physical.null_count() == self.len() && other.physical.null_count() == other.len() {
36            let len = self.len();
37            self.set_lengths(other);
38            new_chunks(&mut self.physical.chunks, &other.physical().chunks, len);
39            return Ok(());
40        }
41
42        if self.is_enum() {
43            self.physical_mut().append(other.physical())?;
44        } else {
45            let mut new_self = call_categorical_merge_operation(self, other, CategoricalAppend)?;
46            std::mem::swap(self, &mut new_self);
47        }
48        Ok(())
49    }
50
51    pub fn append_owned(&mut self, other: Self) -> PolarsResult<()> {
52        // @TODO: Move the implementation to append_owned and make append dispatch here.
53        self.append(&other)
54    }
